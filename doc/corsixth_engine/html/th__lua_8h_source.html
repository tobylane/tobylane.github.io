294 template <class T>
295 static T* luaT_testuserdata(lua_State *L, int idx, int mt_idx, bool required = true)
296 {
297  // Turn mt_idx into an absolute index, as the stack size changes.
298  if(mt_idx > LUA_REGISTRYINDEX && mt_idx < 0)
299  mt_idx = lua_gettop(L) + mt_idx + 1;
300 
301  void *ud = lua_touserdata(L, idx);
302  if(ud != nullptr && lua_getmetatable(L, idx) != 0)
303  {
304  while(true)
305  {
306  if(lua_equal(L, mt_idx, -1) != 0)
307  {
308  lua_pop(L, 1);
309  return (T*)ud;
310  }
311  // Go up one inheritance level, if there is one.
312  if(lua_type(L, -1) != LUA_TTABLE)
313  break;
314  lua_rawgeti(L, -1, 1);
315  lua_replace(L, -2);
316  }
317  lua_pop(L, 1);
318  }
319 
320  if (required)
321  {
322  const char *msg = lua_pushfstring(L, "%s expected, got %s", luaT_classinfo<T>::name(), luaL_typename(L, idx));
323  luaL_argerror(L, idx, msg);
324  }
325  return nullptr;
326 }
