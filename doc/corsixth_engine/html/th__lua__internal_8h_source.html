83  const char * sCurrentClassName = name; \
84  int iCurrentClassMT = pState->aiMetatables[mt]; \
85  lua_settop(pState->L, pState->iTop); \

89  /* Set the __gc metamethod to C++ destructor */ \
90  luaT_pushcclosure(pState->L, luaT_stdgc<typnam, luaT_environindex>, 0); \
91  lua_setfield(pState->L, iCurrentClassMT, "__gc"); \
92  /* Set the depersist size */ \
