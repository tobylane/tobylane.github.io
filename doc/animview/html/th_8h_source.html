23 /*
24  Note that this file contains similar functionality to th.h / th_gfx.h from
25  the main CorsixTH project. It is reimplemented rather than reused for good
26  reasons:
27  1) Animations need to be rendered onto a wxWidgets canvas rather than onto
28  an SDL or DirectX canvas.
29  2) Modifications and experimentation can be performed on this version
30  before being implemented on the game copy, resulting in better code
31  when the game version is implemented.
32  3) Simplicity rather than speed is the aim of this copy, so it will render
33  slower than the game copy, but will be easier to understand and make
34  changes to.
35 */

102  inline void set(int iLayer, int iID)
103  {
104  if(0 <= iLayer && iLayer < 13 && 0 <= iID && iID < 32)
105  m_iMask[iLayer] |= (1 << iID);
106  }
107 
108  void clear();
109 
110  inline void clear(int iLayer, int iID)
111  {
112  if(0 <= iLayer && iLayer < 13 && 0 <= iID && iID < 32)
113  m_iMask[iLayer] &= ~(1 << iID);
114  }
115 
116  inline bool isSet(int iLayer, int iID) const
117  {
118  if(0 <= iLayer && iLayer < 13 && 0 <= iID && iID < 32)
119  return (m_iMask[iLayer] & (1 << iID)) != 0;
120  else
121  return false;
122  }
123  inline bool isSet(int iLayer) const
124  {
125  if(0 <= iLayer && iLayer < 13)
126  for(int iId = 0; iId < 32; ++iId)
127  {
128  if((m_iMask[iLayer] & (1 << iId)) != 0)
129  return true;
130  }
131  return false;
132  }
133 
134 protected:
135  uint32_t m_iMask[13];
136 };

211  {

218 
219  wxFile oFile(sFilename);
220  if(!oFile.IsOpened())
221  return false;
222  size_t iLen = oFile.Length();
223  unsigned char* pBuffer = new unsigned char[iLen];
224  oFile.Read(pBuffer, iLen);
225  if(memcmp(pBuffer, "RNC\001", 4) == 0)
226  {
227  pBuffer = Decompress(pBuffer, iLen);
228  if(!pBuffer)
229  {
230  return false;
231  }
232  }
233  pArray = (T*)pBuffer;
234  iCount = iLen / sizeof(T);
235  return true;
236  }
